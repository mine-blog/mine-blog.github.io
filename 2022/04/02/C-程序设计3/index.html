对象数组与对象指针。
所谓对象数组是指每一个数组元素都是对象的数组，若一个类有若干个对象，我们把这一系列的对象用一个数组来存放。对象数组的元素是对象，不仅具有数据成员，而且还有函数成员。

#include <iostream>
#include <cmath>
using namespace std;
//class类
class exam
{
private:
	int x, y;
public:
	exam(int n)
	{
		x = n;
	}
	int getx()
	{
		return x;
	}
	int caly()
	{
		y = x * x;
		return y;
	}
};
int main()
{
	exam ob1[4] = { 11,22,33,44 };
	for (int i = 0; i < 4; i++)
	{
		cout << ob1[i].getx() << endl;
		cout << ob1[i].caly() << endl;
	}
	return 0;
}

上面也就是直接用了对象数组，创建了几个元素，可以用来灵活调用。
在设计构类的构造函数时要充分考虑到对象数组元素初始化的需要，当各个元素的初始值相同时，可以在类中定义不带参数的构造函数或带有默认参数值的构造函数，当各个元素对象初值要求为不同时，需要定义带参数的构造函数。

#include <iostream>
#include <cmath>
using namespace std;
//class类
class exam
{
private:
	int x;
public:
	exam()
	{
		x = 123;
	}
	exam(int i)
	{
		x = i;
	}
	int getx()
	{
		return x;
	}
};
int main()
{
	exam ob1[4] = { 11,22,33,44 };
	exam ob2[4] = { 55,66 };
	for (int i = 0; i < 4; i++)
	{
		cout << ob1[i].getx() << ' ';
	}
	cout << endl;
	for (int i = 0; i < 4; i++)
	{
		cout << ob2[i].getx() << ' ';
	}
	return 0;
}

上述对数组元素进行了初始化。

#include <iostream>
#include <cmath>
using namespace std;
//class类
class Complex
{
private:
	double real;
	double imag;
public:
	Complex(double r = 0.0, double i = 0.0) :real(r), imag(i)//利用了成员初始化列表进行了初始化
	{}
	~Complex()//三个对象数组元素被删除，因此调用三次析构函数
	{
		cout << "调用了析构函数。" << endl;
	}
	double absComplex()
	{
		double t;
		t = pow(real, 2) + pow(imag, 2);
		return sqrt(t);
	}
};
int main()
{
	Complex com[3] =
	{
		Complex(1.2,2.3),
		Complex(3,4),
		Complex(6,8)
	};
	for (int i = 0; i < 3; i++)
		cout << "复数的绝对值是：" << com[i].absComplex() << endl;
	return 0;
}

对象指针。每一个对象在初始化后都会在内存中占有一定的空间，因此既可以通过对象名访问指针，也可以通过对象地址访问一个对象。对象指针就是用于存放对象地址的变量。

#include <iostream>
#include <cmath>
using namespace std;
//class类
class exe
{
private:
	int x;
public:
	void setx(int a)
	{
		x = a;
	}
	int showx()
	{
		return x;
	}
};
int main()
{
	exe ob;
	exe* p;//p是指针
	p = &ob;
	ob.setx(2);
	//下面这三种表达是一样的效果
	cout << "x是：" << ob.showx() << endl;
	cout << "用指针访问x是：" << p->showx() << endl;
	cout << "用指针访问x是：" << (*p).showx() << endl;
	return 0;
}

用对象指针访问数组。对象指针不仅能引用单个对象，也能访问对象数组。

#include <iostream>
#include <cmath>
using namespace std;
//class类
class exe
{
private:
	int x;
public:
	void setx(int a)
	{
		x = a;
	}
	int showx()
	{
		return x;
	}
};
int main()
{
	exe ob[2];
	exe* p;
	p = ob;//指针指向数组首地址
	ob[0].setx(3);
	ob[1].setx(5);
	cout << "输出：" << p->showx() << endl;
	p++;//当指针加1或减1时，它总是指向其基本类型中相邻的一个元素，对象指针也是如此。
	cout << "输出：" << p->showx() << endl;
	return 0;
}

this指针。当定义了一个类的若干对象后，系统会为每一个对象分配存储空间。如果一个类包含了数据成员和成员函数，就要分别为数据和函数的代码分配存储空间。按照通常的思路，如果用一个类定义了5个对象，那么就应该为这5个对象的数据和函数代码分配存储空间。事实上，给对象赋值就是给对象的数据成员赋值，不同对象的存储单元中存放的数据通常是不相同的，而不同对象的函数代码是相同的，不论调用哪一个对象的成员函数，其实调用的都是相同内容的代码。因此没有必要为每一个对象都开辟存储成员函数的空间。
实际上，C++的编译系统只用了一段空间来存放这个共同的函数代码段，在调用各对象的成员函数时，都去调用这个公用的函数代码。因此每个对象的存储空间都是该对象数据成员所占用的存储空间，而不包括成员函数代码所占用的空间，函数代码是存储在函数对象空间之外的。
每个对象都有属于自己的数据成员，但是所有对象的成员函数代码却合用一份。

C++为成员函数提供了一个名字为this的指针，这个指针称为自引用指针。每当创建一个对象时，系统就把this指针初始化为指向该对象，即this指针的值是当前调用成员函数的对象的起始地址。

#include <iostream>
#include <cmath>
using namespace std;
//class类
class A
{
private:
	int x;
public:
	A(int a)
	{
		x = a;
	}
	void disp()
	{
		cout << "this = " << this << " when x= " << this->x << endl;
	}
};
int main()
{
	A a(1), b(2);
	a.disp();
	b.disp();
	return 0;
}

可以看出随着对象的改变，this指针的值也发生了变化。

向参数传递对象。对象可以作为参数传递给函数，其方法与传递基本类型的变量相同。在向函数传递对象时，是通过传值调用传递给函数的，即单向传递，只由实参传给形参，而不能由形参传给实参。因此函数中对对象的任何修改均不影响调用该函数的对象（实参）本身。
意思就是传参只能在函数体内保留，不会更改对象的真正数据。

#include <iostream>
using namespace std;

class Tr
{
private:
	int i;
public:
	Tr(int n)
	{
		i = n;
	}
	void seti(int n)
	{
		i = n;
	}
	int geti()
	{
		return i;
	}
};

void sqr_it(Tr ob)
{
	ob.seti(ob.geti() * ob.geti());
	cout << "在函数内，形参对象的数据成员的值为：" << ob.geti() << endl;
}

int main()
{
	Tr ob1(10);
	cout << "函数外的数据成员的值为：" << ob1.geti() << endl;
	sqr_it(ob1);
	cout << "函数外的数据成员的值为：" << ob1.geti() << endl;
	return 0;
}

使用对象指针作为函数参数，使用对象指针作为函数参数可以实现”传址调用”，即在函数调用时使实参变量与形参变量指向同一内存地址，在函数调用过程中，对形参对象指针所指对象值的改变也会同样影响实参的值。
效果如下所示，更改了对象的确定值。

#include <iostream>
using namespace std;

class Tr
{
private:
	int i;
public:
	Tr(int n)
	{
		i = n;
	}
	void seti(int n)
	{
		i = n;
	}
	int geti()
	{
		return i;
	}
};

void sqr_it(Tr *ob)
{
	ob->seti(ob->geti() * ob->geti());
	cout << "在函数内，形参对象的数据成员的值为：" << ob->geti() << endl;
}

int main()
{
	Tr ob1(10);
	Tr* ob2 = &ob1;
	cout << "函数外的数据成员的值为：" << ob2->geti() << endl;
	sqr_it(ob2);
	cout << "函数外的数据成员的值为：" << ob1.geti() << endl;
	return 0;
}

使用对象引用作为函数参数，使用对象引用作为函数参数不但具有对象指针用作函数参数的优点，而且对象引用作函数参数更简单、更直接。示例如下：

#include <iostream>
using namespace std;

class Tr
{
private:
	int i;
public:
	Tr(int n)
	{
		i = n;
	}
	void seti(int n)
	{
		i = n;
	}
	int geti()
	{
		return i;
	}
};

void sqr_it(Tr &ob)
{
	ob.seti(ob.geti() * ob.geti());
	cout << "在函数内，形参对象的数据成员的值为：" << ob.geti() << endl;
}

int main()
{
	Tr ob1(10);
	Tr& ob2 = ob1;
	cout << "函数外的数据成员的值为：" << ob2.geti() << endl;
	sqr_it(ob2);
	cout << "函数外的数据成员的值为：" << ob1.geti() << endl;
	return 0;
}
